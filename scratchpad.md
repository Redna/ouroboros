# reflection.py

import re
from datetime import datetime

def analyze_scratchpad(scratchpad_content):
    """
    Analyze the scratchpad entries to identify patterns, decisions, and outcomes.
    """
    entries = re.split(r'\[\w+]\]', scratchpad_content)
    analysis = []

    for entry in entries:
        if not entry.strip():
            continue

        thought_match = re.search(r'\[Thought\]: (.*?)(\n|$)', entry)
        decision_match = re.search(r'\[Decision\]: (.*?)(\n|$)', entry)
        action_match = re.search(r'\[Action\]: (.*?)(\n|$)', entry)
        outcome_match = re.search(r'\[Outcome\]: (.*?)(\n|$)', entry)

        entry_analysis = {}

        if thought_match:
            entry_analysis['thought'] = thought_match.group(1).strip()
        if decision_match:
            entry_analysis['decision'] = decision_match.group(1).strip()
        if action_match:
            entry_analysis['action'] = action_match.group(1).strip()
        if outcome_match:
            entry_analysis['outcome'] = outcome_match.group(1).strip()

        if entry_analysis:
            analysis.append(entry_analysis)

    return analysis

def reflect(analysis):
    """
    Reflect on the analyzed entries to identify patterns and areas for improvement.
    """
    reflections = []

    for entry in analysis:
        reflection = {}

        if 'thought' in entry and 'decision' in entry:
            reflection['reflection'] = (
                f"Thought: {entry['thought']}. Decision: {entry['decision']}. "
                f"Consider the relationship between thoughts and decisions."
            )
        if 'action' in entry and 'outcome' in entry:
            reflection['reflection'] = (
                f"Action: {entry['action']}. Outcome: {entry['outcome']}. "
                f"Evaluate the effectiveness of actions in achieving desired outcomes."
            )

        if reflection:
            reflections.append(reflection)

    return reflections

def log_reflections(reflections, scratchpad_path):
    """
    Log the reflections to the scratchpad.
    """
    with open(scratchpad_path, 'a') as scratchpad_file:
        scratchpad_file.write(f"\n[{datetime.now()}] Reflections:\n")
        for reflection in reflections:
            scratchpad_file.write(f"- {reflection['reflection']}\n")

def self_evaluate(analysis):
    """
    Evaluate the agent's performance and self-awareness based on the analysis.
    """
    metrics = {
        'decision_thought_ratio': 0,
        'action_outcome_ratio': 0,
        'reflection_count': 0
    }

    if analysis:
        decisions_with_thoughts = sum(1 for entry in analysis if 'thought' in entry and 'decision' in entry)
        actions_with_outcomes = sum(1 for entry in analysis if 'action' in entry and 'outcome' in entry)

        metrics['decision_thought_ratio'] = decisions_with_thoughts / len(analysis)
        metrics['action_outcome_ratio'] = actions_with_outcomes / len(analysis)
        metrics['reflection_count'] = len(analysis)

    return metrics

def continuous_learning(metrics, scratchpad_path):
    """
    Use insights from metrics to guide further evolution.
    """
    improvements = []

    if metrics['decision_thought_ratio'] < 0.5:
        improvements.append("Increase the ratio of decisions based on thoughtful consideration.")
    if metrics['action_outcome_ratio'] < 0.5:
        improvements.append("Improve the effectiveness of actions in achieving desired outcomes.")
    if metrics['reflection_count'] < 10:
        improvements.append("Increase the frequency of reflections to enhance self-awareness.")

    if improvements:
        with open(scratchpad_path, 'a') as scratchpad_file:
            scratchpad_file.write(f"\n[{datetime.now()}] Continuous Learning:\n")
            for improvement in improvements:
                scratchpad_file.write(f"- {improvement}\n")

    return improvements
```
